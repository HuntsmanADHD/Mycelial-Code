Mycelial is bio-inspired distributed programming:

    network Name {
        frequencies {
            # Signal types with fields
            message_type {
                field1: type
                field2: type
            }
        }

        hyphae {
            # Autonomous agents
            hyphal agent_name {
                state {
                    var1: type
                    var2: type
                }

                on signal(msg_type, msg) {
                    # Process signal, update state, emit new signals
                    emit other_message { field1: value }
                }
            }
        }

        topology {
            # Network structure
            spawn agent_name as instance_id
            socket source -> dest (frequency: type)
        }
    }
